# ターミナルを起動
# Ctrl + J

# 仮想環境の構築
# conda create -n flask python=3.9

# 仮想環境の起動
# conda activate flask

# 仮想環境にFlaskライブラリをインストール
# conda install flask

# Pythonの対話モードを起動
# python
# -> import flask で Error が発生しなければ成功
# インストールが問題なければ Ctrl + D で対話モードを終了

# ローカルサーバーの終了
# Ctrl + C

# Git Command
# 初期化:		git init
# 連携:			git remote add origin (GitHub上でコピーしたURL)
# 連携確認:		git remote -v
# コミット:		git add .
# コメント付与:	git commit -m "コメント"
# プッシュ:		git push origin master
